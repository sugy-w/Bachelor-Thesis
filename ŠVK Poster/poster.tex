\headerbox{Motivation}{name=motivation, column=0, row=0, span=3}{
  \begin{itemize}[noitemsep,topsep=0pt,parsep=0pt,partopsep=0pt, leftmargin=10pt]
   \item Noisy and high-frequency data generated on the stock exchange markets have been point of interest for decades.
   \item Being able to trust-worthy predict day-to-day changes of stock assets would be extremely beneficial for the industry -- more informed decision in portfolio management, strategy evaluations, etc.
   \item There are quite a few studies involing both classification and regression machine learning (ML) approaches to predict developments on the market; however, many of them apply ML techniques only on very limited number of \cite{ref1,ref2} or on stock indices \cite{ref3}. There is no comprehensive study on whole market segment.  
   \item What is more is lack of studies applying prediction modelling in portfolio strategy optimization.
   \\
   \item This study evaluates prediction capabilities of selected ML and portfolio optimization frameworks on segment of \textbf{Information technology} (IT) among 500 biggest US companies by market cap
   during period between September 2017 and January 2026.
  \end{itemize}  
}

\headerbox{Features for ML modelling}{name=vmeas, span=3, below=motivation, column=0}{
  \begin{itemize}[noitemsep,topsep=0pt,parsep=0pt,partopsep=0pt, leftmargin=10pt]
    \item The amount of variables, which are directly or indirectly generated with day-to-day market running is enormous and qualified selecting is necessary. Selection methodology in this study is based of \textit{a priori} expectations of link between market and particular feature.  
    \\
    \item Among \textbf{stock indices}, S\&P 500 and NASDAQ Composite were chosen for their presumable prediction valuabality.
    \item From the set of raw goods -- \textbf{commodities} -- traded on the market, gold and crude oil were selected. 
    \item Lastly, data regarding \textbf{foreign exchange market} -- exchange rates of selected major currencies (with american dollar), were considered.
    \\
    \item Apart from these, data quantifying \textbf{volatility on the market} and price developments of selected \textbf{US Treasury Bonds} were used.
  \end{itemize}

\headerbox{Rolling window foundations}{name=2ddiag, span=3, below=vmeas, above=foottext, column=0}{
  \begin{itemize}[noitemsep,topsep=0pt,parsep=0pt,partopsep=0pt, leftmargin=10pt]
    \item We adopted \textbf{rolling window} framework for next-day closing price prediction of stocks. Data from trading days $t-q, t-(q-1), \ldots, t-1$ are employed as predictors for the closing price on day $t$. 
  \end{itemize}
  \vspace{0.5cm}
  \hspace*{8pt} Suppose time interval, where prediction applied, is bounded by time points $T_1$ and $T_2$ (suppose $T_2 - T_1 = M$). On this interval reside two frames.
  Multivariate $\mathbf{X} \in \mathbb{R}^{M \times F}$, where $F$ is the number of used features -- values of selected variables, shifted backwards up to $q = 9$ days. This means
  that $\mathbf{X}[t]$ (the $t$-th row of $\mathbf{X}$) represents feature data available through days $t-1, t-2, \ldots, t-9$.

\headerbox{Rolling window implementation}{name=resolution, span=3, row=0, column=3}{
  \begin{center}
  \begin{minipage}{0.9\linewidth}
\begin{algorithmic}
\State \texttt{ \# We assume $w \gets 1$}
\State \texttt{ \# The prediction target day is $t+p+1$}
\State \texttt{ \# We have available data up to day $t+p$}
\State \texttt{Set look-back window length $p$}
\State \texttt{Retrieve the feature variables' values into $\mathbf{X}$}
\\
\For{each stock in database}
    \State Retrieve closing prices into the target $Y$
    \For{$t = T_1$ to $T_2 - (p + 1)$}
        \State $\mathbf{X}_{\text{train}} \gets \mathbf{X}[t : t+p]$
        \State $Y_{\text{train}} \gets Y[t : t+p]$
        \State $\mathbf{X}_{\text{predict}} \gets \mathbf{X}[t+p+1]$
        \State Train model $\hat{f}$ on $(\mathbf{X}_{\text{train}}, Y_{\text{train}})$
        \State Generate prediction: $\hat{f}(\mathbf{X}_{\text{predict}})$
    \EndFor
\EndFor
\end{algorithmic}
\end{minipage}
\end{center}
